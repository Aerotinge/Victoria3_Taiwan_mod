han_claim_china = {
	is_shown = {
		any_primary_culture = { has_discrimination_trait = han_chinese }
		NOT = {
			ROOT = c:CHI
			has_variable = claim_china_var
		}
	}

	possible = {
		exists = c:CHI
		NOT = {
			is_subject_of = c:CHI
		}
		c:CHI = { relations:root <= relations_threshold:poor }	
	}

	when_taken = {
		set_variable = {
			name = claim_china_var
			value = yes
		}
		trigger_event = {
			id = claim_china.1
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		base = 100
	}
}

taiwan_unclaim_china = {
	is_shown = {
		exists = c:FRS
		c:FRS = ROOT
		NOT = {
			has_variable = unclaim_china_var
		}
	}

	possible = {
		any_state = {
			has_claim_by = root
			OR = {
				region = sr:region_north_china
				region = sr:region_south_china
				region = sr:region_manchuria
			}
			NOR = {
				owner = root
				state_region = s:STATE_FORMOSA
			}
		}
	}

	when_taken = {
		set_variable = {
			name = unclaim_china_var
			value = yes
		}
		trigger_event = {
			id = claim_china.2
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		base = 0
	}
}

claim_taiwan_native = {
	is_shown = {
		s:STATE_FORMOSA = {
			any_scope_state = {
				owner = {
					is_country_type = decentralized
				}
			}
			any_scope_state = {
				owner = ROOT
			}
		}
		NOR = {
			is_country_type = decentralized
			has_claim = s:STATE_FORMOSA
		}
	}

	possible = {
	}

	when_taken = {
		s:STATE_FORMOSA = {
			add_claim = ROOT
		}
	}

	ai_chance = {
		base = 100
	}
}
